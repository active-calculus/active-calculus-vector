var('s,t')
param(s,t)=(s,2*cos(t),2*sin(t))
b=[0,5,0,2*pi]
plot22=parametric_plot3d(param,(s,b[0],b[1]),(t,b[2],b[3]),color='blue',opacity=.85)
plot22+=point3d(param(s=0,t=0),color='black',size=10)
plot22+=point3d(param(s=1,t=0),color='black',size=10)
plot22+=point3d(param(s=0,t=1),color='black',size=10)
plot22+=point3d(param(s=2,t=3),color='black',size=10)
show(plot22)
